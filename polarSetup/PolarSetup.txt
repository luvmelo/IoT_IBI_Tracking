import asyncio
from bleak import BleakScanner

async def scan():
    print("Scanning for 15 seconds — keep H10 worn with WET electrodes...")
    devices_and_adv = await BleakScanner.discover(
        timeout=15.0, return_adv=True
    )
    
    print(f"\nFound {len(devices_and_adv)} BLE devices:\n")
    
    polar_found = []
    for address, (device, adv) in devices_and_adv.items():
        name = device.name or adv.local_name or ""
        manuf = adv.manufacturer_data
        services = adv.service_uuids
        rssi = adv.rssi
        
        is_polar = (
            "Polar" in name or 
            107 in manuf or
            any("180d" in s.lower() for s in services)
        )
        
        marker = " *** POLAR/HR ***" if is_polar else ""
        display_name = name if name else "(no name)"
        print(f"  RSSI={rssi:4d} | {display_name:25s} | {address[:8]}...{marker}")
        
        if is_polar:
            polar_found.append((address, device, adv))
    
    print(f"\n{'='*60}")
    if polar_found:
        print(f"Found {len(polar_found)} Polar/HR device(s):")
        for addr, dev, adv in polar_found:
            print(f"   Name: {dev.name or adv.local_name}")
            print(f"   Address: {addr}")
            print(f"   Services: {adv.service_uuids}")
    else:
        print("No Polar H10 detected.")
        print("  - Is the strap WORN with WET electrodes?")
        print("  - Is your PHONE disconnected from H10?")
        print("  - Try removing the H10 pod from strap, then snap it back on")

asyncio.run(scan())
